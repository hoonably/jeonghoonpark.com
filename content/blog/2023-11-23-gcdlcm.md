```python
def gcd(a, b):
    while b > 0:
        a, b = b, a % b
    return a
```

</div><div dir="auto" style="display:contents"><ul class="bulleted-list" id="331451cf-7b79-800e-a093-eed9154f2485"><li style="list-style-type:disc">최소공배수 (Lowest Common Multiple, LCM)</li></ul></div><div dir="auto" style="display:contents">

```python
def lcm(a, b):
    return a * b // gcd(a, b)
```

</div><div dir="auto" style="display:contents"><hr id="331451cf-7b79-80b1-acc1-ece1095fd375"/></div><div dir="auto" style="display:contents"><h3 class="" id="331451cf-7b79-801b-8be3-e3f7c4b17375">자바 (Java)</h3></div><div dir="auto" style="display:contents"><ul class="bulleted-list" id="331451cf-7b79-80ad-8d64-f1efd7ee1473"><li style="list-style-type:disc">최대공약수 (Greatest Common Divisor, GCD)</li></ul></div><div dir="auto" style="display:contents">

```java
public static int gcd(int a, int b) {
    if (b == 0) {
      return a;
    }
    return gcd(b, a % b);
}
```

</div><div dir="auto" style="display:contents"><ul class="bulleted-list" id="331451cf-7b79-800f-840d-de6bc3ea35e6"><li style="list-style-type:disc">최소공배수 (Lowest Common Multiple, LCM)</li></ul></div><div dir="auto" style="display:contents">

```java
public static int lcm(int a, int b) {
    int gcd = gcd(a, b);
    return (a * b) / gcd;
}
```

public static int gcd(int a, int b) {
    if (b == 0) { // 나머지가 0 : 나누어떨어진다면 마지막
      return a;  // 나머지였던 a를 리턴하여 재귀함수를 종료한다.
    }
    return gcd(b, a % b); // 나누어떨어지지 않는다면 재귀함수를 계속 진행한다.
}

```java
import java.util.Scanner;

public class Main {

  // 이 문제에서는 입력값이 매우 크므로 long을 사용해야 하므로 int를 long으로 바꿔준다.

	public static long gcd(long a, long b) {
	    if (b == 0) {
	      return a;
	    }
	    return gcd(b, a % b);
	}

	public static long lcm(long a, long b) {
	    long gcd = gcd(a, b);
	    return (a * b) / gcd;
	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		long a = sc.nextLong();
		long b = sc.nextLong();
		sc.close();

		System.out.println(lcm(a, b));
	}
}
```

</div><div dir="auto" style="display:contents"><hr id="331451cf-7b79-8069-b135-fc9d786d978f"/></div><div dir="auto" style="display:contents"><h3 class="" id="331451cf-7b79-806c-b148-ffc0c602d327"><strong>2. 일반 for문을 사용한 풀이 (알고리즘 X)</strong></h3></div><div dir="auto" style="display:contents">

```java
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		long a = sc.nextLong();
		long b = sc.nextLong();
		long gcd = 1;
		for(int j=2; j<=100000000;) { //2부터 1억까지 공약수의 소인수가 가능한 것을 하나하나 확인
			if(a % j == 0 && b % j == 0) { //동시에 나눠지면 (공약수의 소인수)
				a /= j;	//a도 공약수의 소인수로 나누고
				b /= j;	//b도 공약수의 소인수로 나누고
				gcd *= j; // j 는 최대공약수의 소인수이므로 곱하면서 저장
			}
			else {
				j++; //안나눠질때만 다음으로 넘어감. 2^4 * 3^2 이런 식의 소인수분해가 가능하기 때문
			}
		}
		long	lcm = a * b * gcd; //여기서 a b는 서로소이므로 셋을 곱하면 최소공배수이다.
		System.out.println(gcd);
		System.out.println(lcm);
		sc.close();
	}

}
```
