```python
import sys


def q_check(x):
    # 이미 기록된 0 ~ N 범위 보기
    for i in range(N):

        # 본인 단계 제외
        if i==x:
            continue

        # 아직 안쓴거는 대각선 영향갈수있으니 제외
        if row[i]==0:
            continue

        # row (가로)체크
        # 체크하려는 row == 기록된 row(q_list[i]) 일때 False
        if row[x] == row[i]:
            return False
        # 대각선 체크
        # 체크하려는 row 와 이미 기록된 row 의 차가
        # 체크하려는 column 과 기록된 column 의 차랑 같다면 대각선이니 False
        if abs(row[x]-row[i]) == abs(x-i): # abs() = 절댓값 함수
            return False
    # 두 경우 다 피해가면 True
    return True


def n_queen(n):
    # 끝까지 완성한 경우
    if n==N:
        print(*row)
        quit() # 아예 종료

    # 이미 써있는경우 바로 다음단계로
    if row[n]!=0:
        n_queen(n+1)
        return

    # 1부터 N까지 넣어보기
    for i in range(1,N+1):
        # 해당 숫자를 사용하지 않았을 때만 넣기
        if visited[i] == False:
            temp = row[n]
            row[n] = i
            if q_check(n):
                visited[i]=True
                n_queen(n+1)
                visited[i]=False
            row[n] = temp

N = int(sys.stdin.readline())
row = list(map(int, sys.stdin.readline().split()))
visited = [False] * (N+1)

# 이미 놓아진 퀸의 값
for i in range(N):
    if row[i] != 0:
        visited[row[i]] = True

n_queen(0) # 0번째 줄부터 함수 실행
print(-1)
```

</div><div dir="auto" style="display:contents"><hr id="331451cf-7b79-8050-91bc-d9a92fc0d043"/></div><div dir="auto" style="display:contents"><h3 class="" id="331451cf-7b79-8006-9531-f769ef3df684">더 빠른 풀이법</h3></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-80b7-9914-d74d967e1fa5">이전의 글을 보면 알 수 있지만, 확인하는 과정에서 for문을 사용하지 않고, 바로 확인 할 수 있다.</p></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-8023-ba90-d02616a51364">이전의 글을 보지 않았다면 아래로 내려가 설명을 보자.</p></div><div dir="auto" style="display:contents">

```python
import sys

def backtracking(n):

    # 다 도달했을 때
    if n==N:
        print(*arr)
        quit()

    # 이미 써있는경우 바로 다음단계로
    if arr[n] != 0:
        backtracking(n + 1)
        return

    for i in range(N):
        # 놓을 수 있으면
        if row[i] and x1[i+n] and x2[i+((N-1)-n)]:

            arr[n] = i+1
            row[i]=False # 가로줄 제거
            x1[i+n]=False # 오른쪽 위 방향 대각선 제거
            x2[i+((N-1)-n)]=False # 오른쪽 아래 방향 대각선 제거

            # 자식 노드로 이동
            backtracking(n+1)

            # 백트래킹
            row[i]=True
            x1[i+n]=True
            x2[i+((N-1)-n)]=True
            arr[n] = 0

N = int(sys.stdin.readline())
arr = list(map(int, sys.stdin.readline().split()))

row = [True]*N # 가로
x1 = [True]*(2*N) # 제일 왼쪽이 인덱스인 우상향 대각선
x2 = [True]*(2*N) # 제일 오른쪽이 인덱스인 우하향 대각선

for i in range(N):
    if arr[i] != 0:
        row[arr[i]-1] = False  # 가로줄 제거
        x1[arr[i]-1 + i] = False  # 오른쪽 위 방향 대각선 제거
        x2[arr[i]-1 + ((N - 1) - i)] = False  # 오른쪽 아래 방향 대각선 제거

backtracking(0)
print(-1)
```

</div><div dir="auto" style="display:contents"><hr id="331451cf-7b79-8080-983f-d46dd6ebf309"/></div><div dir="auto" style="display:contents"><h3 class="" id="331451cf-7b79-80cd-9a89-c47be74522d5">row[] : 가로 줄</h3></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-8021-a0f4-fa66daae1511">row는 간단하게 가로에 겹치는게 있으면 안되므로, 해당 row를 사용했으면 False로 바꿔주기만 하면 된다.</p></div><div dir="auto" style="display:contents"><hr id="331451cf-7b79-8093-ad3d-c502dd570584"/></div><div dir="auto" style="display:contents"><h3 class="" id="331451cf-7b79-802e-8e09-f24a4ba91665">x2[] : 우하향 대각선</h3></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-8072-936c-d957de596bd4"><code>x2</code>는 <strong>오른쪽 아래</strong>로 가는 대각선을 뜻하고, 총 <strong>2*N</strong> 개가 나올 수 있다.</p></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-80a4-a55f-dd31317a31b8">각 대각선들은 <strong>가장 오른쪽에 갔을때</strong> index를 기준으로 삼았다.</p></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-80e4-97a3-ead67ddbcbb2">그림은 0, 1 단계를 마치고 <strong>2단계</strong>의 for문에서 <strong>i=3</strong>일때 우하향 대각선이 겹치지 않는지를 보는 것이다.</p></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-8089-9762-fe19b0fb4331">그렇다면 n 단계의 for문의 i번째 row일때 생기는 <strong>우하향</strong> 대각선은 <strong>맨 오른쪽</strong>으로 가면</p></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-80d8-ad38-eb5fd7ddf0e4"><code>i + (마지막 단계 - 지금 단계)</code> 가 된다.</p></div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-80dd-bd7f-d6ad7288fc16">그러므로 놓았을때는</p></div><div dir="auto" style="display:contents">

```python
x2[i+((N-1)-n)] = False 로 바꿔주어 대각선이 사용되었음을 나타내면 된다.
```

</div><div dir="auto" style="display:contents"><p class="" id="331451cf-7b79-808f-b5c2-cc7b4f37d3fd">들어갈 수 있는지 체크할때는</p></div><div dir="auto" style="display:contents">

```python
x2[i+((N-1)-n)] 가 True면 겹치지 않고, False면 겹쳐서 불가능 한것이다.
```

